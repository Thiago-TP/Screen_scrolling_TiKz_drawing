%% goombas e koopas
\foreach \line in {0, 1, 2, 3} {
	\foreach \col in {0, 1} {			
		\foreach [count=\i from 0] \color in \arrayHalfEnemiesOne {
			\slantedPixel{\color}{\i}{
				(up left.center) ++ 
				(0, -4*\tileLen*\line) ++ 
				(\angle:160*\col)}{160}{1}
		}
		\foreach [count=\i from 0] \color in \arrayHalfEnemiesTwo {
			\slantedPixel{\color}{\i}{
				(up left.center) ++ (0, -\tileLen) ++
				(0, -4*\tileLen*\line) ++ 
				(\angle:160*\col)}{160}{1}
		}
	}
}
