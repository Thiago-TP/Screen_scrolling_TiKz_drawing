%% goombas e koopas
\foreach \line in {0, 1, 2, 3} {
	\foreach \col in {0, 1, 2} {			
		\foreach [count=\i from 0] \color in \arrayEnemiesOne {
			\slantedPixel{\color}{\i}{
				(tilemapStart.center) ++ 
				(0, -4*\tileLen*\line) ++ 
				(\angle:160*\col)}{160}{.25}
		}
		\foreach [count=\i from 0] \color in \arrayEnemiesTwo {
			\slantedPixel{\color}{\i}{
				(tilemapStart.center) ++ (0, -\tileLen) ++
				(0, -4*\tileLen*\line) ++ 
				(\angle:160*\col)}{160}{.25}
		}
	}
}

}
